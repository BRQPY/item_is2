def isCyclicConnected(adj: dict, s, V,
                      visited: dict):
#DIVIDER
    parent = {}
    for key, value in adj.items():
        parent[key] = -1
#DIVIDER
    q = []
#DIVIDER
    visited[s] = True
    q.append(s)

    while q != []:
#DIVIDER
        u = q.pop()
#DIVIDER
        for v in adj[u]:
            if not visited[v]:
                visited[v] = True
                q.append(v)
                parent[v] = u
            elif parent[u] != v:
                return True

    return False

def isCyclicDisconnected(adj: dict, V):
#DIVIDER
    visited = {}
    for key, value in adj.items():
        visited[key] = False

    for key, value in adj.items():
        if not visited[key] and \
                isCyclicConnected(adj, key, V, visited):
            return True
    return False
